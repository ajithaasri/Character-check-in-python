# Character-check-in-python
ch = input("Enter a character: ")

if (ch == 'A' or ch == 'a' or
    ch == 'E' or ch == 'e' or
    ch == 'I' or ch == 'i' or
    ch == 'O' or ch == 'o' or
    ch == 'U' or ch == 'u'):
    print(ch, "is a Vowel")
else:
    print(ch, "is a Consonant")

OUTPUT:
1. Enter a character: a
a is a Vowel

2. Enter a character: b
b is a Consonant
